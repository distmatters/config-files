% Create the Letterhead and To/From Block
\renewcommand{\maketitle}{\makememotitle}
\newcommand\makememotitle{
	$if(memologo)$
	{ % Create With Logo
	\begin{minipage}[t]{1\columnwidth}%
		\begin{flushright}
			\vspace{-0.6in}
			$logo$
			\vspace{0.5in}
		\par\end{flushright}%
	\end{minipage}
	}
	$endif$
	% To, From, Subject Block
	\begin{description}[noitemsep]
		$if(to)$\item [{\sffamily To:}] \tabto{24 pt} $to$$endif$
		$if(author)$\item [{\sffamily From:}] \tabto{24 pt} $author$$endif$
		$if(title)$\item [{\sffamily Subject:}] \tabto{24 pt} $title$$endif$
		$if(date)$\item [{\sffamily Date:}] \tabto{24 pt} $date$$endif$
	\end{description} \vspace{-1em}
	\decorativeline
}
